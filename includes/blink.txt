<PUBLIC:METHOD NAME="doBlink" />
<PUBLIC:ATTACH event="oncontentready" handler="Init" />

<SCRIPT>

function doBlink() {
  this.style.visibility = this.style.visibility=="" ? "hidden" : ""
}

function Init() {
  if (!speed) speed=1000
  setInterval(uniqueID+".doBlink()",speed)
}
</SCRIPT>
